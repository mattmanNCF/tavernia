```python
# app/agents/providers/mock.py
# Mock AgentProvider for testing — no network calls, no real AI SDK imports.
# Imported by get_agent_provider() when AGENT_PROVIDER=mock in .env / test env.
from dataclasses import dataclass, field
from datetime import datetime, timezone


@dataclass
class AgentResult:
    content: str
    domain: str
    timestamp: datetime
    metadata: dict = field(default_factory=dict)


class Provider:
    """Mock provider. Returns deterministic responses without any AI API call."""

    def __init__(self, model: str = "mock-model"):
        self.model = model

    async def _call_provider(self, messages: list[dict]) -> str:
        # Deterministic: echo back the user message content.
        for msg in reversed(messages):
            if msg.get("role") == "user":
                return f"[mock response] {msg['content'][:80]}"
        return "[mock response] no user message found"

    async def run(self, task: str, domain: str, username: str | None = None) -> AgentResult:
        """Mimic AgentProvider.run() without loading real context files or writing wiki."""
        raw = await self._call_provider([{"role": "user", "content": task}])
        return AgentResult(
            content=raw,
            domain=domain,
            timestamp=datetime.now(timezone.utc),
            metadata={"model": self.model, "mock": True},
        )
```

```python
# This ensures no real AI SDK is ever imported during tests.

os.environ.setdefault("AGENT_MODEL", "mock-model")
os.environ.setdefault("SECRET_KEY", "test-secret-key-not-for-production")
os.environ.setdefault("DATABASE_URL", "postgresql+asyncpg://tavernia:tavernia@localhost:5432/tavernia_test")

# --- DB session fixture (async) ---
@pytest_asyncio.fixture
async def db_session():
    """
    Provide an AsyncSession connected to the test database.
    Rolls back after each test (no persistent side effects).
    Import lazily so app modules aren't loaded before env vars are set.
    """
    from sqlalchemy.ext.asyncio import create_async_engine, async_sessionmaker, AsyncSession
    test_url = os.environ["DATABASE_URL"]
    engine = create_async_engine(test_url, echo=False)
    factory = async_sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)
    async with factory() as session:
        yield session
        await session.rollback()
    await engine.dispose()


# --- HTTP client fixture ---
@pytest_asyncio.fixture
async def client():
    """
    Provide an httpx AsyncClient against the FastAPI app.
    Import lazily so env vars are set before app import.
    """
    from app.main import app
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as ac:
        yield ac

    return Provider(model="mock-model")
```

```python
# app/main.py
# FastAPI application factory.
#
# CRITICAL INITIALIZATION ORDER (from RESEARCH.md Pitfall 2):
# 1. init_plugin_manager(PLUGINS_DIR) is called at MODULE IMPORT TIME (line ~20)
#    — BEFORE app = FastAPI() and BEFORE the lifespan handler.
# 2. The lifespan handler fires pm.hook.on_startup(registry=helper) inside an
#    AsyncSession context, then awaits helper.flush(session) to commit DB rows.
# 3. app = FastAPI(lifespan=lifespan) is called AFTER the PluginManager is initialized.
#
# Violating this order causes Pitfall 2: plugins miss on_startup because the
# PluginManager wasn't ready when on_startup fired.
from contextlib import asynccontextmanager
from pathlib import Path

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

from app.core.plugin_manager import init_plugin_manager, PluginRegistryHelper
from app.api.health import router as health_router
from app.api.auth import router as auth_router
from app.api.rbac import router as rbac_router

# Resolve plugins directory relative to this file
PLUGINS_DIR = Path(__file__).parent.parent / "plugins"

# STEP 1: Build PluginManager EAGERLY at module import time.
# This discovers all plugins in PLUGINS_DIR before FastAPI app is created.
pm = init_plugin_manager(PLUGINS_DIR)


@asynccontextmanager
async def lifespan(app: FastAPI):
    """
    FastAPI lifespan handler (replaces deprecated @app.on_event).
    Startup: fire on_startup for all plugins, flush DB registrations.
    Shutdown: (future) plugin teardown hooks.
    """
    # STEP 2: Fire on_startup inside a DB session so PluginRegistryHelper can flush.
    from app.database import AsyncSessionLocal
    async with AsyncSessionLocal() as session:
        helper = PluginRegistryHelper()
        # pm.hook.on_startup is synchronous — plugins call helper.register() synchronously
        pm.hook.on_startup(registry=helper)
        # Flush queued registrations to plugin_registry table
        await helper.flush(session)

    yield  # Application is now running

    # Shutdown: (future) pm.hook.on_shutdown() for cleanup
    pass


# STEP 3: Create FastAPI app AFTER PluginManager is initialized.
app = FastAPI(
    title="Tavernia",
    version="0.1.0",
    description="Modular open-source hospitality OS",
    lifespan=lifespan,
)

# CORS — allow Vite dev server (localhost:5173) during development.
# In production, nginx serves frontend and API on the same origin — CORS not needed.
app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:5173"],
    allow_credentials=True,  # Required for cookie-based auth
    allow_methods=["*"],
    allow_headers=["*"],
)

# Mount routers
app.include_router(health_router)   # GET /health
app.include_router(auth_router)     # POST /auth/login, /auth/logout, GET /auth/me
app.include_router(rbac_router)     # GET /chef/dashboard, /gm/dashboard, /accountant/dashboard
```

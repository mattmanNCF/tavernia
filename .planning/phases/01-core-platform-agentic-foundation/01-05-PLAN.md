```python
# app/core/hooks.py
# pluggy hookspecs for Tavernia. All plugins implement these interfaces.
# Phase 1 defines on_startup and on_event only.
# Domain-specific hooks (on_invoice_received, etc.) added in later phases.
import pluggy

hookspec = pluggy.HookspecMarker("tavernia")
hookimpl = pluggy.HookimplMarker("tavernia")


class TaverniaSpec:
    @hookspec
    def on_startup(self, registry) -> None:
        """
        Called once at app startup. Plugin registers itself via registry.register().
        registry is a PluginRegistryHelper instance — calls are synchronous but
        inserts are queued and flushed by the lifespan handler after all hooks fire.
        """

    @hookspec
    def on_event(self, event_name: str, payload: dict) -> None:
        """
        Called when a versioned event is emitted on the event bus.
        Naming convention: domain.action.v1 (e.g. hello.world.v1).
        Plugins must NOT call emit_event on another plugin's internal events (CORE-03).
        """
```

```python
# app/core/plugin_manager.py
# Plugin discovery, PluginManager singleton, and PluginRegistryHelper.
#
# CRITICAL: build_plugin_manager() must be called at module import time (before FastAPI app is created).
# If called inside a lifespan/startup handler, plugins miss on_startup (Pitfall 2 from RESEARCH.md).
import importlib.util
import pluggy
from pathlib import Path

from app.core.hooks import TaverniaSpec

# Module-level singleton — set by init_plugin_manager()
_pm: pluggy.PluginManager | None = None


class PluginRegistryHelper:
    """
    Passed as the 'registry' argument to on_startup hooks.
    on_startup is SYNCHRONOUS (pluggy limitation). register() is async (DB insert).
    Solution: queue pending inserts synchronously, flush them after all hooks fire.
    The lifespan handler calls await helper.flush(session) after pm.hook.on_startup(registry=helper).
    """

    def __init__(self):
        self._pending: list[dict] = []

    def register(self, name: str, version: str, description: str = "") -> None:
        """Queue a plugin registration. Flushed to DB after all on_startup hooks complete."""
        self._pending.append({"name": name, "version": version, "description": description})

    async def flush(self, session) -> None:
        """Upsert all queued registrations into the plugin_registry table."""
        if not self._pending:
            return
        from sqlalchemy.dialects.postgresql import insert as pg_insert
        from app.core.schema import PluginRegistry

        for row in self._pending:
            stmt = (
                pg_insert(PluginRegistry)
                .values(**row)
                .on_conflict_do_update(
                    index_elements=["name"],
                    set_={"version": row["version"], "description": row["description"]},
                )
            )
            await session.execute(stmt)
        await session.commit()
        self._pending.clear()


def build_plugin_manager(plugins_dir: Path) -> pluggy.PluginManager:
    """
    Discover all plugins in plugins_dir and return a configured PluginManager.
    Each plugin must have a plugin.py file. Registers any @hookimpl found in the module.
    """
    pm = pluggy.PluginManager("tavernia")
    pm.add_hookspecs(TaverniaSpec)

    if not plugins_dir.exists():
        return pm

    for plugin_path in plugins_dir.iterdir():
        if not plugin_path.is_dir():
            continue
        plugin_file = plugin_path / "plugin.py"
        if not plugin_file.exists():
            continue

        spec = importlib.util.spec_from_file_location(
            f"plugins.{plugin_path.name}.plugin",
            plugin_file,
        )
        mod = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(mod)
        # Register any class/object with @hookimpl methods
        # pluggy looks for hookimpl markers on all attributes
        for attr_name in dir(mod):
            attr = getattr(mod, attr_name)
            if isinstance(attr, type) and attr_name.endswith("Plugin"):
                pm.register(attr())
                break
        else:
            pm.register(mod)

    return pm


def init_plugin_manager(plugins_dir: Path) -> pluggy.PluginManager:
    """Initialize the module-level _pm singleton. Called once at import time in main.py."""
    global _pm
    _pm = build_plugin_manager(plugins_dir)
    return _pm


def get_plugin_manager() -> pluggy.PluginManager:
    """Return the initialized plugin manager singleton."""
    if _pm is None:
        raise RuntimeError("Plugin manager not initialized. Call init_plugin_manager() first.")
    return _pm
```

```python
# app/core/event_bus.py
# Thin wrapper around pluggy's on_event hook dispatch.
# emit_event() is the ONLY way plugins communicate cross-plugin (CORE-03).
# No plugin may subscribe to another plugin's internal events — all events go through this bus.


def emit_event(event_name: str, payload: dict) -> None:
    """
    Dispatch a versioned event to all registered on_event hookimpls.

    Naming convention: domain.action.v1 (e.g. hello.world.v1, invoice.received.v1)
    Plugins must NOT call emit_event with another plugin's internal event names.

    Args:
        event_name: Versioned event name (domain.action.v1)
        payload: Event data dict
    """
    from app.core.plugin_manager import get_plugin_manager
    pm = get_plugin_manager()
    pm.hook.on_event(event_name=event_name, payload=payload)
```

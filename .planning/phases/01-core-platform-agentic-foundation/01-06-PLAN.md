```python
# app/api/auth.py
# Auth routes: POST /auth/login, POST /auth/logout, GET /auth/me
# JWT stored in httpOnly SameSite=Strict cookie (CONTEXT.md locked decision).
# 15-minute access token; no refresh token in Phase 1.
import uuid
from datetime import datetime, timedelta, timezone

import jwt
from fastapi import APIRouter, Depends, HTTPException, Response, status, Cookie
from fastapi.security import OAuth2PasswordRequestForm
from passlib.context import CryptContext
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.deps import DBSession
from app.core.schema import User, Role, UserRole
from app.settings import settings

router = APIRouter(prefix="/auth", tags=["auth"])
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

ACCESS_TOKEN_EXPIRE_MINUTES = 15


def _create_access_token(user_id: str, username: str, role: str) -> str:
    """Create a signed JWT with sub, username, role, exp claims."""
    expire = datetime.now(timezone.utc) + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    payload = {
        "sub": user_id,
        "username": username,
        "role": role,
        "exp": expire,
    }
    return jwt.encode(payload, settings.secret_key, algorithm="HS256")


async def _get_user_role(db: AsyncSession, user_id: uuid.UUID) -> str | None:
    """Look up role name via user_roles join table. Returns None if no role assigned."""
    stmt = (
        select(Role.name)
        .join(UserRole, UserRole.role_id == Role.id)
        .where(UserRole.user_id == user_id)
        .limit(1)
    )
    result = await db.execute(stmt)
    row = result.scalar_one_or_none()
    return row


@router.post("/login")
async def login(
    response: Response,
    form_data: OAuth2PasswordRequestForm = Depends(),
    db: DBSession = Depends(),
):
    """Authenticate user; set httpOnly JWT cookie on success."""
    stmt = select(User).where(User.username == form_data.username)
    result = await db.execute(stmt)
    user = result.scalar_one_or_none()

    if not user or not pwd_context.verify(form_data.password, user.hashed_password):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect username or password",
        )

    if not user.is_active:
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Account inactive")

    role = await _get_user_role(db, user.id)
    if not role:
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="No role assigned")

    token = _create_access_token(
        user_id=str(user.id),
        username=user.username,
        role=role,
    )

    response.set_cookie(
        key="access_token",
        value=token,
        httponly=True,
        samesite="strict",
        max_age=ACCESS_TOKEN_EXPIRE_MINUTES * 60,
        secure=False,  # Set True in production behind HTTPS/nginx
    )
    return {"message": "logged in", "role": role}


@router.post("/logout")
async def logout(response: Response):
    """Clear the access_token cookie."""
    response.delete_cookie("access_token")
    return {"message": "logged out"}


@router.get("/me")
async def me(access_token: str = Cookie(None)):
    """Return current user info from JWT without DB lookup."""
    if not access_token:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Not authenticated")
    try:
        payload = jwt.decode(access_token, settings.secret_key, algorithms=["HS256"])
        return {"username": payload["username"], "role": payload["role"], "sub": payload["sub"]}
    except jwt.ExpiredSignatureError:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Token expired")
    except jwt.InvalidTokenError:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token")
```

```python
# app/api/deps.py
# Shared FastAPI dependencies: DB session, current user, role enforcement.
from typing import Annotated

import jwt
from fastapi import Cookie, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession

from app.database import get_db
from app.settings import settings

DBSession = Annotated[AsyncSession, Depends(get_db)]


async def get_current_user(access_token: str = Cookie(None)) -> dict:
    """
    Extract and verify JWT from httpOnly cookie.
    Returns payload dict with sub, username, role keys.
    Raises 401 if cookie missing, expired, or invalid.
    """
    if not access_token:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Not authenticated",
        )
    try:
        payload = jwt.decode(access_token, settings.secret_key, algorithms=["HS256"])
        return payload
    except jwt.ExpiredSignatureError:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Token expired")
    except jwt.InvalidTokenError:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token")


def require_role(*roles: str):
    """
    FastAPI dependency factory. Enforces role-based access control.
    Usage: @router.get("/chef/dashboard", dependencies=[Depends(require_role("chef"))])

    Uses user_roles join table via JWT role claim (set at login from DB lookup).
    The role claim is embedded in the JWT — no DB query on every request.
    """
    async def checker(user: dict = Depends(get_current_user)) -> dict:
        if user.get("role") not in roles:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"Role '{user.get('role')}' is not authorized for this route",
            )
        return user
    return checker
```

```python
# scripts/seed_db.py
# One-time DB seeder. Creates the 3 built-in roles and 1 test user per role.
# Run AFTER alembic upgrade head:
#   python scripts/seed_db.py
#
# Test credentials seeded:
#   chef_user / chef_pass123     (role: chef)
#   gm_user / gm_pass123         (role: gm)
#   accountant_user / acct_pass123 (role: accountant)
import asyncio
import uuid

from passlib.context import CryptContext
from sqlalchemy import select
from sqlalchemy.ext.asyncio import create_async_engine, async_sessionmaker, AsyncSession

from app.core.schema import User, Role, UserRole

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

# Use sync-friendly import of settings so script is runnable outside Docker
import os
DATABASE_URL = os.environ.get(
    "DATABASE_URL",
    "postgresql+asyncpg://tavernia:tavernia@localhost:5432/tavernia",
)

ROLES = ["chef", "gm", "accountant"]

TEST_USERS = [
    {"username": "chef_user",       "email": "chef@tavernia.local",       "password": "chef_pass123",  "role": "chef"},
    {"username": "gm_user",         "email": "gm@tavernia.local",         "password": "gm_pass123",    "role": "gm"},
    {"username": "accountant_user", "email": "acct@tavernia.local",        "password": "acct_pass123",  "role": "accountant"},
    # Second chef user — proves AGENT-06 (distinct wiki profiles for two chefs)
    {"username": "chef_user2",      "email": "chef2@tavernia.local",      "password": "chef_pass123",  "role": "chef"},
]


async def seed():
    engine = create_async_engine(DATABASE_URL, echo=True)
    factory = async_sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)

    async with factory() as session:
        # Create roles (idempotent)
        role_map: dict[str, uuid.UUID] = {}
        for role_name in ROLES:
            stmt = select(Role).where(Role.name == role_name)
            result = await session.execute(stmt)
            existing = result.scalar_one_or_none()
            if existing:
                role_map[role_name] = existing.id
            else:
                role = Role(name=role_name, description=f"Built-in role: {role_name}")
                session.add(role)
                await session.flush()
                role_map[role_name] = role.id

        # Create test users + assign roles
        for u in TEST_USERS:
            stmt = select(User).where(User.username == u["username"])
            result = await session.execute(stmt)
            existing = result.scalar_one_or_none()
            if existing:
                print(f"User {u['username']} already exists, skipping")
                continue

            user = User(
                username=u["username"],
                email=u["email"],
                hashed_password=pwd_context.hash(u["password"]),
                is_active=True,
            )
            session.add(user)
            await session.flush()

            user_role = UserRole(
                user_id=user.id,
                role_id=role_map[u["role"]],
                location_id=None,  # Single-location install — no location filtering
            )
            session.add(user_role)

        await session.commit()
        print("Seed complete.")

    await engine.dispose()


if __name__ == "__main__":
    asyncio.run(seed())
```

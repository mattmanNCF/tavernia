```python
# app/agents/provider.py
# AgentProvider — abstract base class. Contains ZERO AI SDK imports.
# Concrete providers live in app/agents/providers/{name}.py.
# This module must remain SDK-free so importing it never triggers provider library loading.
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime, timezone


@dataclass
class AgentResult:
    """Structured output from any AgentProvider.run() call."""
    content: str
    domain: str
    timestamp: datetime
    metadata: dict = field(default_factory=dict)


class AgentProvider(ABC):
    """
    Abstract AI provider. Concrete implementations in app/agents/providers/.
    No provider SDK (anthropic, openai, google, ollama) may be imported in this file
    or in any module that imports this file.

    Subclasses must implement _call_provider(messages) -> str.
    The run() template method handles context loading and wiki write-back.
    """

    def __init__(self, model: str):
        self.model = model

    @abstractmethod
    async def _call_provider(self, messages: list[dict]) -> str:
        """Send messages to the underlying AI provider, return raw text response."""
        ...

    async def run(
        self,
        task: str,
        domain: str,
        username: str | None = None,
    ) -> AgentResult:
        """
        Execute a task in a domain context.

        Steps:
        1. Load modules/{domain}/context.md
        2. Load wiki/{domain}/*.md KB sections
        3. Optionally load wiki/users/{username}.md if username provided
        4. Call _call_provider with assembled messages
        5. Append structured outcome to wiki/{domain}/outcomes.md
        6. Create/update wiki/users/{username}.md if username provided
        7. Return AgentResult

        Args:
            task: Natural language task description
            domain: Domain name, maps to modules/{domain}/context.md
            username: Optional. If provided, loads and updates user-specific wiki profile.
                     Profile keyed by username, NOT by role — two chefs get two separate files.
        """
        # Import lazily to keep provider.py SDK-free at module load
        from app.agents.loader import load_context, load_wiki_context
        from app.agents.wiki import append_outcome, create_or_update_user_profile

        context_md = load_context(domain)
        wiki_ctx = load_wiki_context(domain)

        system_content = f"{context_md}\n\n--- Wiki KB ---\n\n{wiki_ctx}"
        if username:
            from app.agents.wiki import load_user_profile
            user_profile = load_user_profile(username)
            system_content += f"\n\n--- User Profile ({username}) ---\n\n{user_profile}"

        messages = [
            {"role": "system", "content": system_content},
            {"role": "user", "content": task},
        ]

        raw = await self._call_provider(messages)
        result = AgentResult(
            content=raw,
            domain=domain,
            timestamp=datetime.now(timezone.utc),
            metadata={"model": self.model},
        )

        await append_outcome(domain, task, result)
        if username:
            await create_or_update_user_profile(username, domain, task, result)

        return result
```

```python
# app/agents/loader.py
# Dynamic provider loading + context/wiki file loading for AgentProvider.run().
# get_agent_provider() uses importlib so no SDK is imported at module level.
import importlib
from pathlib import Path
from app.settings import settings

# Project root — resolves relative to this file's location
PROJECT_ROOT = Path(__file__).parent.parent.parent


def load_context(domain: str) -> str:
    """
    Load modules/{domain}/context.md and return its content.
    AGENT-02: each domain module ships a context.md describing the domain.
    Returns empty string with warning if file not found (non-fatal).
    """
    context_path = PROJECT_ROOT / "modules" / domain / "context.md"
    if not context_path.exists():
        return f"[context.md not found for domain: {domain}]"
    return context_path.read_text(encoding="utf-8")


def load_wiki_context(domain: str) -> str:
    """
    Load all .md files from wiki/{domain}/ and concatenate them.
    AGENT-03: agents read the relevant KB section on each invocation.
    Returns empty string if wiki domain directory does not exist.
    """
    wiki_dir = PROJECT_ROOT / "wiki" / domain
    if not wiki_dir.exists():
        return f"[wiki/{domain}/ not found — no KB context loaded]"
    parts = []
    for md_file in sorted(wiki_dir.glob("*.md")):
        parts.append(md_file.read_text(encoding="utf-8"))
    return "\n\n---\n\n".join(parts) if parts else f"[wiki/{domain}/ is empty]"


def get_agent_provider():
    """
    Dynamically load the configured AgentProvider.
    AGENT-01: no SDK imported at module level — importlib loads only the requested provider.
    AGENT_PROVIDER env var controls which provider class is loaded.
    Example: AGENT_PROVIDER=mock loads app/agents/providers/mock.py class Provider.
    """
    provider_name = settings.agent_provider  # e.g., "mock", "anthropic", "openai"
    try:
        module = importlib.import_module(f"app.agents.providers.{provider_name}")
    except ModuleNotFoundError:
        raise ValueError(
            f"Unknown AGENT_PROVIDER='{provider_name}'. "
            f"Create app/agents/providers/{provider_name}.py with a Provider class."
        )
    cls = getattr(module, "Provider")
    return cls(model=settings.agent_model)
```

```python
# app/agents/wiki.py
# Wiki KB read/write helpers for AgentProvider.run().
# AGENT-03: read wiki/{domain}/*.md
# AGENT-04: append structured outcome after each task
# AGENT-05/06: create and update wiki/users/{username}.md profiles
import re
from datetime import datetime, timezone
from pathlib import Path
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from app.agents.provider import AgentResult

PROJECT_ROOT = Path(__file__).parent.parent.parent
WIKI_ROOT = PROJECT_ROOT / "wiki"


def _make_frontmatter(domain: str, created: datetime | None = None) -> str:
    """Generate YAML frontmatter block. All three fields (created, last_updated, domain) are mandatory."""
    now = datetime.now(timezone.utc).isoformat()
    c = (created or datetime.now(timezone.utc)).isoformat()
    return f"---\ndomain: {domain}\ncreated: {c}\nlast_updated: {now}\n---\n\n"


def _update_last_updated(file_path: Path) -> None:
    """Update last_updated field in existing YAML frontmatter."""
    content = file_path.read_text(encoding="utf-8")
    now = datetime.now(timezone.utc).isoformat()
    content = re.sub(
        r"^last_updated: .+$",
        f"last_updated: {now}",
        content,
        count=1,
        flags=re.MULTILINE,
    )
    file_path.write_text(content, encoding="utf-8")


async def append_outcome(domain: str, task: str, result: "AgentResult") -> None:
    """
    AGENT-04: Append structured markdown outcome entry to wiki/{domain}/outcomes.md.
    Creates the file (with YAML frontmatter) if it does not exist.
    Entry format (from CONTEXT.md):
      ## [timestamp] task summary
      ### What was done
      ### What was learned
      ### Confidence
      ### Edge cases
    """
    domain_dir = WIKI_ROOT / domain
    domain_dir.mkdir(parents=True, exist_ok=True)
    outcomes_file = domain_dir / "outcomes.md"

    ts = result.timestamp.isoformat()
    task_summary = task[:80].replace("\n", " ")

    entry = (
        f"\n## [{ts}] {task_summary}\n\n"
        f"### What was done\n{result.content[:500]}\n\n"
        f"### What was learned\n(agent-populated)\n\n"
        f"### Confidence\n{result.metadata.get('confidence', 'medium')}\n\n"
        f"### Edge cases\n(agent-populated)\n\n"
    )

    if not outcomes_file.exists():
        outcomes_file.write_text(
            _make_frontmatter(domain) + f"# {domain} — Agent Outcomes\n",
            encoding="utf-8",
        )

    with outcomes_file.open("a", encoding="utf-8") as f:
        f.write(entry)

    _update_last_updated(outcomes_file)


def load_user_profile(username: str) -> str:
    """Load wiki/users/{username}.md content, or return empty-profile placeholder."""
    profile_path = WIKI_ROOT / "users" / f"{username}.md"
    if not profile_path.exists():
        return f"[No profile yet for user: {username}]"
    return profile_path.read_text(encoding="utf-8")


async def create_or_update_user_profile(
    username: str,
    domain: str,
    task: str,
    result: "AgentResult",
) -> None:
    """
    AGENT-05/06: Create or update wiki/users/{username}.md.
    Profile is keyed by username — NOT by role.
    Two users sharing 'chef' role produce SEPARATE files: wiki/users/alice.md, wiki/users/bob.md.
    """
    users_dir = WIKI_ROOT / "users"
    users_dir.mkdir(parents=True, exist_ok=True)
    profile_path = users_dir / f"{username}.md"

    ts = datetime.now(timezone.utc).isoformat()
    interaction_note = (
        f"\n## [{ts}] Interaction in domain: {domain}\n"
        f"Task: {task[:80]}\n"
        f"Outcome summary: {result.content[:200]}\n"
    )

    if not profile_path.exists():
        # Create new profile with mandatory frontmatter
        profile_path.write_text(
            _make_frontmatter("users")
            + f"# User Profile: {username}\n\n"
            + "## Preferences\n(accumulated by agents over time)\n\n"
            + "## Workflow Patterns\n(accumulated by agents over time)\n\n"
            + "## Interaction History\n"
            + interaction_note,
            encoding="utf-8",
        )
    else:
        with profile_path.open("a", encoding="utf-8") as f:
            f.write(interaction_note)
        _update_last_updated(profile_path)
```
